function storageGet(k,f){try{const d=localStorage.getItem(k);return d?JSON.parse(d):f;}catch(e){return f;}}

function storageSet(k,v){localStorage.setItem(k,JSON.stringify(v));}


if(!localStorage.getItem('candidates'))storageSet('candidates',{"Jayaprakash":0,"Vijay Manohar":0,"Priyaharsha":0,"Mercy Sukeerthi":0});

if(!localStorage.getItem('voters'))storageSet('voters',{});

if(!localStorage.getItem('notifications'))storageSet('notifications',[]);

if(!localStorage.getItem('notifSeen'))storageSet('notifSeen',false);


function show(id){

  document.querySelectorAll('section').forEach(s=>s.style.display='none');

  document.getElementById(id).style.display='block';

  if(id==='home'){renderResults();renderNotifications();markNotifSeen();}

  if(id!=='login'){logout();}

}


async function registerFingerprint(){

  const roll=document.getElementById('regRoll').value.trim();

  const msg=document.getElementById('fingerMsg');

  const voters=storageGet('voters',{});

  if(!roll){msg.style.color='red';msg.textContent='Enter roll number.';return;}

  if(voters[roll]){msg.style.color='red';msg.textContent='Roll already registered.';return;}

  const deviceKey=localStorage.getItem('deviceFingerprint');

  if(deviceKey){msg.style.color='red';msg.textContent='This device is already registered for another voter.';return;}

  try{

    const pk={challenge:new Uint8Array(32),rp:{name:"College Election 2025"},

      user:{id:new Uint8Array(16),name:roll,displayName:roll},

      pubKeyCredParams:[{alg:-7,type:"public-key"}],

      authenticatorSelection:{userVerification:"preferred"}};

    const credential=await navigator.credentials.create({publicKey:pk});

    if(!credential)throw new Error('Fingerprint not captured');

    const credId=btoa(String.fromCharCode(...new Uint8Array(credential.rawId)));

    voters[roll]={approved:false,voted:false,credId};

    storageSet('voters',voters);

    localStorage.setItem('deviceFingerprint',credId);

    msg.style.color='green';

    msg.textContent='✅ Fingerprint registered successfully. Await admin approval.';

  }catch(e){msg.style.color='red';msg.textContent='Fingerprint registration failed or cancelled.';}

}


async function loginFingerprint(){

  const roll=document.getElementById('loginRoll').value.trim();

  const msg=document.getElementById('loginMsg');

  const voters=storageGet('voters',{});

  if(!roll||!voters[roll]){msg.style.color='red';msg.textContent='Voter not found';return;}

  if(!voters[roll].approved){msg.style.color='red';msg.textContent='Not approved yet';return;}

  try{

    const pk={challenge:new Uint8Array(32),

      allowCredentials:[{type:"public-key",id:Uint8Array.from(atob(voters[roll].credId),c=>c.charCodeAt(0))}],

      userVerification:"preferred"};

    const assertion=await navigator.credentials.get({publicKey:pk});

    if(!assertion)throw new Error('Fingerprint verification failed');

    const deviceKey=localStorage.getItem('deviceFingerprint');

    if(deviceKey!==voters[roll].credId){msg.style.color='red';msg.textContent='Fingerprint mismatch. Use same device.';return;}

    msg.textContent='';document.getElementById('loginForm').style.display='none';

    document.getElementById('voteArea').style.display='block';buildVoteButtons(roll);

  }catch(e){msg.style.color='red';msg.textContent='Fingerprint login failed or cancelled.';}

}


function buildVoteButtons(roll){

  const candidates=storageGet('candidates',{}),c=document.getElementById('voteButtons');

  c.innerHTML='';

  for(const n of Object.keys(candidates)){

    const b=document.createElement('button');b.textContent=n;b.style.display='block';b.style.margin='6px 0';

    b.onclick=()=>castVote(roll,n);c.appendChild(b);

  }

}


function castVote(roll,cand){

  const voters=storageGet('voters',{});if(voters[roll].voted){alert('You already voted');return;}

  voters[roll].voted=true;storageSet('voters',voters);

  const cands=storageGet('candidates',{});cands[cand]=(cands[cand]||0)+1;storageSet('candidates',cands);

  const area=document.getElementById('voteArea');

  area.innerHTML=`<h3 style="color:green;text-align:center;">✅ Thank you for voting for <span style="color:#0a3d62">${cand}</span>!</h3><p style="text-align:center;">Redirecting to home...</p>`;

  setTimeout(()=>{logout();show('home');renderResults();},2500);

}

function logout(){document.getElementById('loginForm').reset();document.getElementById('loginForm').style.display='block';document.getElementById('voteArea').style.display='none';}


document.getElementById('adminLoginForm').addEventListener('submit',e=>{

  e.preventDefault();const u=adminUser.value.trim(),p=adminPass.value.trim(),m=adminMsg;

  if(u==='admin'&&p==='admin'){m.textContent='';adminLoginBox.style.display='none';adminPanel.style.display='block';showPending();}

  else{m.style.color='red';m.textContent='Invalid credentials (use admin/admin)';}

});


function showPending(){

  const v=storageGet('voters',{}),c=adminContent;

  c.innerHTML='<h3>Pending Voters</h3>';const t=document.createElement('table');

  t.innerHTML='<thead><tr><th>Roll</th><th>Approved</th><th>Action</th></tr></thead>';const b=document.createElement('tbody');

  for(const [r,i]of Object.entries(v)){const tr=document.createElement('tr');tr.innerHTML=`<td>${r}</td><td>${i.approved?'✅':'⏳'}</td>`;

    const td=document.createElement('td');if(!i.approved){const a=document.createElement('button');a.textContent='Approve';a.onclick=()=>{i.approved=true;storageSet('voters',v);showPending();};

      const d=document.createElement('button');d.textContent='Reject';d.style.marginLeft='6px';d.onclick=()=>{delete v[r];storageSet('voters',v);showPending();};

      td.appendChild(a);td.appendChild(d);}else td.textContent='-';tr.appendChild(td);b.appendChild(tr);}

  t.appendChild(b);c.appendChild(t);

}


function showAdminResults(){

  const c=storageGet('candidates',{}),con=adminContent;

  con.innerHTML='<h3>Vote Counts</h3>';const t=document.createElement('table');

  t.innerHTML='<thead><tr><th>Candidate</th><th>Votes</th></tr></thead>';const b=document.createElement('tbody');

  for(const[n,v]of Object.entries(c)){b.innerHTML+=`<tr><td>${n}</td><td>${v}</td></tr>`;}t.appendChild(b);con.appendChild(t);

}


function showNotificationPanel(){

  const c=adminContent;

  const notes=storageGet('notifications',[]);

  c.innerHTML='<h3>Manage Notifications</h3>';

  const ta=document.createElement('textarea');ta.style.width='100%';ta.style.height='80px';ta.placeholder='Enter new notification...';

  const btn=document.createElement('button');btn.textContent='Post Notification';btn.onclick=()=>{

    const msg=ta.value.trim();if(!msg)return alert('Enter message');

    notes.unshift({msg,time:new Date().toLocaleString()});storageSet('notifications',notes);

    storageSet('notifSeen',false);renderNotifications();showDot();alert('Notification posted!');ta.value='';

  };

  c.appendChild(ta);c.appendChild(btn);


  if(notes.length){const list=document.createElement('div');list.innerHTML='<h4>Existing Notifications</h4>';

    notes.forEach((n,i)=>{

      const d=document.createElement('div');

      d.innerHTML=`<div style="border:1px solid #ccc;margin:5px 0;padding:6px;border-radius:6px;">

      <strong>${n.time}</strong><br>${n.msg}<br>

      <button onclick="deleteNotification(${i})">Delete</button></div>`;

      list.appendChild(d);

    });

    c.appendChild(list);

  }

}

function deleteNotification(index){

  const notes=storageGet('notifications',[]);notes.splice(index,1);storageSet('notifications',notes);

  alert('Notification deleted!');showNotificationPanel();renderNotifications();

}


function renderNotifications(){

  const notes=storageGet('notifications',[]),d=document.getElementById('notifications');

  if(!notes.length){d.innerHTML='<p>No notifications yet.</p>';return;}

  d.innerHTML=notes.map(n=>`<div><strong>${n.time}</strong><br>${n.msg}</div>`).join('');

}

function showDot(){

  const dot=document.getElementById('notifDot');

  const seen=storageGet('notifSeen',true);

  if(!seen)dot.style.display='inline-block';else dot.style.display='none';

}

function markNotifSeen(){storageSet('notifSeen',true);showDot();}


function adminLogout(){adminPanel.style.display='none';adminLoginBox.style.display='block';adminLoginForm.reset();}

function resetElection(){if(confirm('Reset all data?')){localStorage.clear();location.reload();}}


function renderResults(){

  const c=storageGet('candidates',{}),t=Object.values(c).reduce((a,b)=>a+b,0);

  let h='';for(const[n,v]of Object.entries(c)){const p=t?((v/t)*100).toFixed(1):0;

    h+=`<div><strong>${n}</strong> — ${v} votes (${p}%)</div>

    <div class="barContainer"><div class="bar" style="width:${p}%"></div></div>`;}

  results.innerHTML=h;

}

renderResults();renderNotifications();showDot();
